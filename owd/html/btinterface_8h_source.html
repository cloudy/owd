108 int parseMessage(
109  /* Input */
110  int id ,
111  int len ,
112  unsigned char *messageData ,
113 
114  /* Output */
115  int *node ,
116  int *property ,
117  long *value )
118 {
119  int i;
120  int dataHeader;
121 
122  *node = ADDR2NODE(id);
123  if (*node == -1)
124  fprintf(stderr,"msgID:%x ",id);
125  dataHeader = ((messageData[0] >> 6) & 0x0002) | ((id & 0x041F) == 0x0403) | ((id & 0x41F) == 0x0407);
126  //messageData[0] &= 0x7F;
127  //fprintf(stderr,"Entering parsemessage");
128  switch (dataHeader)
129  {
130  case 3: /* Data is a packed 22-bit position, SET */
131  *value = 0x00000000;
132  *value |= ( (long)messageData[0] << 16) & 0x003F0000;
133  *value |= ( (long)messageData[1] << 8 ) & 0x0000FF00;
134  *value |= ( (long)messageData[2] ) & 0x000000FF;
135 
136  if (*value & 0x00200000) /* If negative */
137  *value |= 0xFFC00000; /* Sign-extend */
138 
139  if((id & 0x041F) == 0x0403)
140  *property = P;
141  else
142  *property = JP;
143 
144  jointPosition[*node] = 0;
145  if(len > 3){
146  jointPosition[*node] |= ( (long)messageData[3] << 16) & 0x003F0000;
147  jointPosition[*node] |= ( (long)messageData[4] << 8 ) & 0x0000FF00;
148  jointPosition[*node] |= ( (long)messageData[5] ) & 0x000000FF;
149 
150  if (jointPosition[*node] & 0x00200000) /* If negative */
151  jointPosition[*node] |= 0xFFC00000; /* Sign-extend */
152  }
153 
154  //fprintf(stderr,"Received packed set property: %d from node: %d value:%d",*property,*node,*value);
155  break;
156  case 2: /* Data is normal, SET */
157  *property = messageData[0] & 0x7F;
158  //fprintf(stderr, "Received property: %d", *property);
159  /* Store the value, second byte of message is zero (for DSP word alignment) */
160  *value = messageData[len-1] & 0x80 ? -1L : 0;
161  for (i = len-1; i >= 2; i--)
162  *value = *value << 8 | messageData[i];
163 
164  //fprintf(stderr, "Received normal set property: %d from node: %d value:%d", *property, *node, *value);
165  //fprintf(stderr,"parsemessage after %d",value);
166  break;
167  case 0: /* Assume firmware request (GET) */
168  *property = -(messageData[0] & 0x7F); /* A negative (or zero) property means GET */
169  *value = 0;
170  //fprintf(stderr, "Received normal get property: %d from node: %d value:%d", *property, *node, *value);
171  break;
172  default:
173  fprintf(stderr, "<Illegal Message Header> %d\n", dataHeader);
174  return(1);
175  }
176  //if (*property != 8) fprintf(stderr,"Value in parsemessage is: %d",*value);
177  return (0);
178 
179 }

181 int getProperty(int bus, int id, int property, long *reply)
182 {
183  int err;
184  unsigned char data[8];
185  int len_in;
186  int id_in;
187  int property_in;
188 
189 
190  // Compile the packet
191  data[0] = (unsigned char)property;
192 
193  // Send the packet
194  err = canSendMsg(bus, NODE2ADDR(id), 1, data, TRUE);
195 
196  if (err) {
197  fprintf(stderr,"getProperty(): canSendMsg error = %d\n", err);
198  return (1);
199  }
200 
201  // Wait for 1 reply
202  err = canReadMsg(bus, &id_in, &len_in, data, TRUE);
203 
204  if(!err) {
205  // Parse the reply
206  err = parseMessage(id_in, len_in, data, &id_in, &property_in, reply);
207 
208 
209  // Check that the id and property match
210  if((id == id_in) && (property == property_in)) {
211  return(0);
212  } else {
213  fprintf(stderr, "getProperty(): returned id or property do not match");
214  fprintf(stderr, "asked for %d from %d but got %d from %d\n",
215  property, id, property_in, id_in);
216  return(1);
217  }
218  } else {
219  fprintf(stderr,"getProperty(): canReadMsg error = %d\n", err);
220  return(1);
221  }
222 }
